#

# scripts/extract-blobs.sh on the stock firmware dump.
#

#   - vendor/lib64/hw/ (HAL implementations)

#   - vendor/firmware/ (firmware blobs)
#   - vendor/etc/ (configuration files)
#   - system_ext/etc/ (system extension configs)
#   - odm/etc/ (ODM configs)
